# HUAWEI HealthKit連携 × 秘書強化 開発仕様書

**作成日**: 2026-03-15
**担当**: リサーチ部・開発部・制作・納品事業部

---

## 要件定義

### 目的・背景

`/company` の秘書が、毎朝ブリーフィング時にオーナーのHUAWEIヘルスバンドから
睡眠・ストレス・心拍データを読み取り、健康状態に応じたスケジュール調整・アドバイスを行う。

「タスク管理秘書」から「心身のコンディションを踏まえた全人的秘書」への進化。

### 対象ユーザー

渡邊カンパニー オーナー（HUAWEI Watch/Band 所持者）

### 機能要件

| # | 機能 | 優先度 |
|---|------|-------|
| 1 | 朝ブリーフィング時に昨夜の睡眠スコアを取得・表示 | ★ MVP |
| 2 | ストレスレベル・安静時心拍数の取得・表示 | ★ MVP |
| 3 | スコアに応じたスケジュール調整提案（睡眠<70点 → 集中タスクを午後へ） | ★ MVP |
| 4 | スコアに応じたアドバイス文の自動生成 | ★ MVP |
| 5 | トークンの自動リフレッシュ（24時間有効期限対応） | ○ Phase 2 |
| 6 | 閾値超え時のDiscordアラート（リアルタイム） | ○ Phase 2 |
| 7 | 週次健康サマリーレポート生成 | △ Phase 3 |

### 非機能要件

- 取得失敗時（API障害・トークン期限切れ）はスキップしてブリーフィングを継続
- 健康データはローカルに保存しない（毎回APIから取得）
- スクリプト実行時間：5秒以内（タイムアウト設定必須）

#### レスポンス → 秘書への受け渡し形式

`fetch_health.py` は以下のJSONをstdoutに出力する：

```json
{
  "date": "2026-03-15",
  "sleep_score": 68,
  "sleep_duration_h": 6.5,
  "stress_level": "medium",
  "resting_hr": 62,
  "steps_yesterday": 7200,
  "advice_flag": "caution",
  "summary": "睡眠スコア68点（やや低め）、ストレス中程度"
}
```

`advice_flag` の定義：

| フラグ | 条件 | 秘書の対応 |
|-------|------|-----------|
| `good` | sleep≥80 かつ stress=low | 通常通りスケジュール |
| `caution` | sleep 60〜79 または stress=medium | 集中タスクを午後にずらす提案 |
| `rest` | sleep<60 または stress=high | 重要タスクを翌日に延期提案 + Discordアラート |

## fetch_health.py 実装

```python
#!/usr/bin/env python3
"""
HUAWEI Health Kit データ取得スクリプト
~/.claude/huawei-health/fetch_health.py
"""

import json
import sys
import os
import datetime
import urllib.request
import urllib.error

CONFIG_PATH = os.path.expanduser("~/.claude/huawei-health/config.json")
API_BASE = "https://health-api.cloud.huawei.com/healthkit/v1"
TIMEOUT = 5  # 秒

def load_config():
    with open(CONFIG_PATH) as f:
        return json.load(f)

def get_yesterday_range():
    """昨日の0:00〜23:59のUnixタイムスタンプ(ms)を返す"""
    today = datetime.date.today()
    yesterday = today - datetime.timedelta(days=1)
    tz = datetime.timezone(datetime.timedelta(hours=9))
    start = datetime.datetime(yesterday.year, yesterday.month, yesterday.day, 0, 0, 0, tzinfo=tz)
    end = datetime.datetime(yesterday.year, yesterday.month, yesterday.day, 23, 59, 59, tzinfo=tz)
    return int(start.timestamp() * 1000), int(end.timestamp() * 1000)

def fetch_data(access_token, start_ms, end_ms):
    payload = {
        "polymerizeWith": [
            {"dataTypeName": "com.huawei.continuous.sleep"},
            {"dataTypeName": "com.huawei.continuous.stress"},
            {"dataTypeName": "com.huawei.point.heart_rate.resting"},
            {"dataTypeName": "com.huawei.continuous.steps.delta"}
        ],
        "startTime": start_ms,
        "endTime": end_ms,
        "groupByTime": {
            "groupPeriod": {
                "timeZone": "+0900",
                "unit": "day",
                "value": 1
            }
        }
    }
    url = f"{API_BASE}/sampleSet:polymerize"
    data = json.dumps(payload).encode()
    req = urllib.request.Request(url, data=data, headers={
        "Authorization": f"Bearer {access_token}",
        "Content-Type": "application/json"
    })
    with urllib.request.urlopen(req, timeout=TIMEOUT) as resp:
        return json.loads(resp.read())

def parse_response(raw):
    """APIレスポンスから健康指標を抽出（実際のレスポンス構造に合わせて調整が必要）"""
    result = {
        "date": str(datetime.date.today() - datetime.timedelta(days=1)),
        "sleep_score": None,
        "sleep_duration_h": None,
        "stress_level": None,
        "resting_hr": None,
        "steps_yesterday": None,
    }

    # ※ 実際のレスポンスフィールド名はAPIドキュメントで確認して調整すること
    groups = raw.get("groups", [])
    for group in groups:
        for dataset in group.get("dataset", []):
            dtype = dataset.get("dataTypeName", "")
            points = dataset.get("point", [])
            if not points:
                continue
            val = points[0].get("values", [{}])[0]

            if "sleep" in dtype:
                result["sleep_score"] = val.get("intVal") or val.get("floatVal")
            elif "stress" in dtype:
                raw_stress = val.get("intVal", 0)
                if raw_stress < 30:
                    result["stress_level"] = "low"
                elif raw_stress < 60:
                    result["stress_level"] = "medium"
                else:
                    result["stress_level"] = "high"
            elif "heart_rate.resting" in dtype:
                result["resting_hr"] = val.get("intVal") or val.get("floatVal")
            elif "steps" in dtype:
                result["steps_yesterday"] = val.get("intVal") or val.get("longVal")

    return result

def judge_flag(data):
    sleep = data.get("sleep_score") or 100
    stress = data.get("stress_level") or "low"
    if sleep < 60 or stress == "high":
        return "rest"
    elif sleep < 80 or stress == "medium":
        return "caution"
    else:
        return "good"

def build_summary(data, flag):
    parts = []
    if data["sleep_score"] is not None:
        parts.append(f"睡眠スコア{data['sleep_score']}点")
    if data["stress_level"]:
        label = {"low": "低め", "medium": "中程度", "high": "高め"}[data["stress_level"]]
        parts.append(f"ストレス{label}")
    if data["resting_hr"] is not None:
        parts.append(f"安静時心拍{data['resting_hr']}bpm")
    if not parts:
        return "データ取得なし"
    return "、".join(parts)

def main():
    try:
        config = load_config()
        start_ms, end_ms = get_yesterday_range()
        raw = fetch_data(config["access_token"], start_ms, end_ms)
        data = parse_response(raw)
        data["advice_flag"] = judge_flag(data)
        data["summary"] = build_summary(data, data["advice_flag"])
        print(json.dumps(data, ensure_ascii=False))
    except FileNotFoundError:
        # config未設定の場合はスキップ（秘書は健康データなしで続行）
        print(json.dumps({"error": "config_not_found", "summary": "（健康データ未設定）"}))
    except Exception as e:
        print(json.dumps({"error": str(e), "summary": "（健康データ取得失敗）"}))

if __name__ == "__main__":
    main()
```
